Tailwind CSS 커스텀 설정

# tailwind.config.js
module.exports = {
    theme: {
    extend: {
        textShadow: {
        'outline': `-2px -2px 0 white, 2px -2px 0 white, 
                    -2px 2px 0 white, 2px 2px 0 white`,
        'outline-thick': `-3px -3px 0 white, 3px -3px 0 white,
                            -3px 3px 0 white, 3px 3px 0 white,
                            -4px 0 0 white, 4px 0 0 white,
                            0 -4px 0 white, 0 4px 0 white`
        }
    }
    },
    plugins: [
    function({ addUtilities }) {
        addUtilities({
        '.text-outline': {
            '-webkit-text-stroke': '2px white',
            'text-stroke': '2px white'
        }
        })
    }
    ]
}
